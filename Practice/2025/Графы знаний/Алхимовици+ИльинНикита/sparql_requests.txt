1. Предметы категории Core, которые стоят больше 4000 золота
PREFIX : <http://dota2.ontology#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>

SELECT ?item ?itemName ?cost WHERE {
    ?item rdf:type :Item ;
          rdfs:label ?itemName ;
          :hasCost ?cost ;
          :hasItemType "Core" .
    FILTER(?cost > 4000)
}
ORDER BY DESC(?cost)

2. Предметы поддержки стоимостью от 500 до 1500 золота
PREFIX : <http://dota2.ontology#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
SELECT ?item ?itemName ?cost ?itemType WHERE {
    ?item rdf:type :Item ;
          rdfs:label ?itemName ;
          :hasCost ?cost ;
          :hasItemType ?itemType .
    FILTER(?cost >= 500 && ?cost <= 1500 && ?itemType = "Support")
}
ORDER BY ?cost
3. Способности с кулдауном больше 60 секунд 
PREFIX : <http://dota2.ontology#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
SELECT ?ability ?abilityName ?hero ?heroName ?cooldown ?manaCost WHERE {
    ?hero rdf:type :Hero ;
          rdfs:label ?heroName ;
          :hasAbility ?ability .
    ?ability rdf:type :Ability ;
             rdfs:label ?abilityName ;
             :hasCooldown ?cooldown ;
             :hasManaCost ?manaCost .
    FILTER(?cooldown >= 60)
}
ORDER BY DESC(?cooldown)
